[Function:ChryVoucherTypes]
	
	Local Formula		: AccessAllowedCreate		: $$Allow:CREATE:$Name
	Local Formula		: AccessAllowedAlter		: $$Allow:ALTER:$Name
	Local Formula		: AccessAllowedPRINT		: $$Allow:ALTER:$Name
	
	010 : Walk Collection	: ChryVoucherTypes
;	020 : Log	: $Name  + " -- " + $$string:@AccessAllowedCreate  + $$String:@AccessAllowedAlter + $$String:@AccessAllowedPRINT
;	030 : Log	: $$Allow:ALTER:$Name;$$Translate:$Name
	060 : End Walk
;	070 : Log			: "--------------"
	

[Function:IsVoucherApplicable]
	Parameter			: VchType			: String
	Variable			: Value				: Number
	010 : Set	: Value	: $$FilterCount:ChryVoucherTypes:ChrVchFilterCondition	
;	011 : Log	: ##VchType+"--"+$$String:##Value
	040 : Do If	: (##Value = 1) : Return	: Yes 
		
	
	

[Collection:ChryVoucherTypes] 
	
	Type		: VoucherType
	Fetch		: Name
	
	Filter		: FltByAlloudAccess 
	
[System:Formulas]
	FltByAlloudAccess	 	: $$Allow:CREATE:$Name OR $$Allow:ALTER:$Name OR $$Allow:ALTER:$Name  
	ChrVchFilterCondition	: ##VchType = $Name
	FilterByCherryUserAccess: $$IsVoucherApplicable:$VoucherTypeName	
	FltByChrryVchTypeMst	: $$IsVoucherApplicable:$Name
