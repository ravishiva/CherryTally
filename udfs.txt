	Parameter	: Line				: Number
	Parameter	: Address			: String
	Parameter	: Type				: String

	Variable	: KeyVar			: String

	010 : List Add		: ByrCnsAddrList				: ##KeyVar
	015 : Set	: IndexVar			: $$ListIndex:ByrCnsAddrList:##KeyVar
	020 : Set	: ByrCnsAddrList[##IndexVar].Line			: ##Line
	030 : Set	: ByrCnsAddrList[##IndexVar].Address		: ##Address
	040 : Set	: ByrCnsAddrList[##IndexVar].Type			: ##Type
	050 : Log	: ##ByrCnsAddrList[##IndexVar].Address  + " -- " + ##ByrCnsAddrList[##IndexVar].Type
